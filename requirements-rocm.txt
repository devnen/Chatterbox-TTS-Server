# requirements-rocm.txt
#
# AMD ROCm dependencies for Chatterbox TTS Server.
#
# IMPORTANT: PyTorch (torch, torchaudio, torchvision, triton) are installed
# separately via requirements-rocm-init.txt to avoid version conflicts.
# The Dockerfile.rocm and start.py handle this automatically.
#
# For manual installation, run these commands in order:
#   pip install -r requirements-rocm-init.txt
#   pip install -r requirements-rocm.txt
#   pip install --no-deps git+https://github.com/devnen/chatterbox-v2.git@master
#
# The --no-deps flag on chatterbox-tts is critical: without it, pip will
# replace the ROCm torch wheels with CPU-only versions from PyPI.
